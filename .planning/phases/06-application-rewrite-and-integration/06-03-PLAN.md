    **tests/FixedTimeSubsystem.h (header-only):**
    ```cpp
    #pragma once
    #include "app/AppSubsystem.h"
    #include "core/Result.h"

    namespace Wayfinder
    {
        class FixedTimeSubsystem final : public AppSubsystem
        {
        public:
            static constexpr float DEFAULT_DT = 1.0f / 60.0f;

            [[nodiscard]] auto Initialise(EngineContext&) -> Result<void> override
            {
                return {};
            }

            void Shutdown() override {}

            void Update() { m_elapsedTime += m_deltaTime; }

            [[nodiscard]] auto GetDeltaTime() const -> float { return m_deltaTime; }
            [[nodiscard]] auto GetElapsedTime() const -> float { return m_elapsedTime; }

            void SetDeltaTime(float dt) { m_deltaTime = dt; }

        private:
            float m_deltaTime = DEFAULT_DT;
            float m_elapsedTime = 0.0f;
        };
    }
    ```

    This is a lightweight AppSubsystem living in test headers, not shipped with the engine. Integration tests register it via a test plugin when they need frame-by-frame time control.

    No RequiredCapabilities, no DependsOn. Header-only for simplicity.
    No CMakeLists.txt entry needed (it's a header included by test files, not a compiled source).
