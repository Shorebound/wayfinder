    #include <doctest/doctest.h>
    #include "app/ApplicationStateMachine.h"
    #include "app/EngineContext.h"

    TEST_SUITE("Application Integration")
    {
        // Minimal test state that counts lifecycle calls
        class CountingState : public IApplicationState
        {
        public:
            int EnterCount = 0, ExitCount = 0, UpdateCount = 0, RenderCount = 0;
            bool RequestStopAfterFrames = false;
            int MaxFrames = 3;

            auto OnEnter(EngineContext& ctx) -> Result<void> override
            { ++EnterCount; return {}; }
            auto OnExit(EngineContext& ctx) -> Result<void> override
            { ++ExitCount; return {}; }
            void OnUpdate(EngineContext& ctx, float dt) override
            {
                ++UpdateCount;
                if (RequestStopAfterFrames and UpdateCount >= MaxFrames)
                    ctx.RequestStop();
            }
            void OnRender(EngineContext& ctx) override { ++RenderCount; }
            auto GetName() const -> std::string_view override { return "CountingState"; }
        };

        TEST_CASE("Headless boot-to-frame cycle completes")
        {
            // 1. Create subsystem registry with FixedTimeSubsystem only (no SDL, no GPU)
            SubsystemRegistry<AppSubsystem> registry;
            registry.Register<FixedTimeSubsystem>({});
            auto manifest = registry.Finalise();
            REQUIRE(manifest.has_value());

            // 2. Wire EngineContext
            EngineContext context;
            context.SetAppSubsystems(&*manifest);

            // 3. Initialise manifest (FixedTimeSubsystem::Initialise)
            CapabilitySet caps; // empty - headless
            auto initResult = manifest->Initialise(context, caps);
            CHECK(initResult.has_value());

            // 4. Configure ASM with CountingState
            ApplicationStateMachine asm;
            asm.AddState<CountingState>();
            asm.SetInitialState(typeid(CountingState));
            auto finaliseResult = asm.Finalise();
            REQUIRE(finaliseResult.has_value());

            context.SetStateMachine(&asm);
            asm.Start(context);

            // 5. Run N frames
            OverlayStack overlayStack;
            context.SetOverlayStack(&overlayStack);
            EventQueue eventQueue;

            constexpr int FRAME_COUNT = 5;
            for (int i = 0; i < FRAME_COUNT and not context.IsStopRequested(); ++i)
            {
                asm.ProcessPending(context);
                auto* state = asm.GetActiveState();
                REQUIRE(state != nullptr);

                overlayStack.ProcessEvents(context, eventQueue);
                state->OnUpdate(context, 0.016f);
                overlayStack.Update(context, 0.016f);
                state->OnRender(context);
                overlayStack.Render(context);
            }

            // 6. Verify
            auto* countingState = dynamic_cast<CountingState*>(asm.GetActiveState());
            REQUIRE(countingState != nullptr);
            CHECK(countingState->EnterCount == 1);
            CHECK(countingState->UpdateCount == FRAME_COUNT);
            CHECK(countingState->RenderCount == FRAME_COUNT);
        }

        TEST_CASE("Stop request terminates frame loop")
        {
            // Similar but CountingState requests stop after 3 frames
            // Verify UpdateCount == 3 (not FRAME_COUNT)
        }

        TEST_CASE("State transition during frame loop")
        {
            // Two states: StateA transitions to StateB
            // Verify OnExit(A) and OnEnter(B) are called
        }

        TEST_CASE("Headless operation without window subsystem")
        {
            // Boot with empty capability set
            // Verify frame loop runs without null pointer dereference
        }
    }
