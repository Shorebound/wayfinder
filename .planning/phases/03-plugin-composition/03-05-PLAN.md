Add to the public section (before Run()):
```cpp
/// Register a plugin or plugin group for v2 composition.
/// Plugins are built in dependency order during Initialise().
/// Per APP-01: This is the sole public API for v2 plugin composition.
template<typename T>
void AddPlugin()
{
    if (not m_builder)
    {
        m_builder = std::make_unique<AppBuilder>();
    }
    m_builder->AddPlugin<T>();
}
```

Add to private members:
```cpp
std::unique_ptr<AppBuilder> m_builder;
std::optional<AppDescriptor> m_appDescriptor;
```

Note: `std::optional<AppDescriptor>` works because AppDescriptor is move-only and default-constructible. It lets Application own the descriptor as a value. The optional is empty until Finalise succeeds.

**4. Update `Application.cpp` Initialise() method:**

In the `Initialise()` method, after project loading and config loading but before the main loop setup, add v2 builder finalisation:

```cpp
// V2 plugin composition
if (m_builder)
{
    // Set project paths for config loading
    if (m_project)
    {
        m_builder->SetProjectPaths(
            m_project->ResolveConfigDir(),
            m_project->ProjectRoot / "saved");
    }

    auto descriptorResult = m_builder->Finalise();
    if (not descriptorResult)
    {
        Log::Error("Application", "AppBuilder::Finalise() failed: {}", descriptorResult.error().GetMessage());
        return std::unexpected(descriptorResult.error());
    }
    m_appDescriptor = std::move(*descriptorResult);
    m_builder.reset(); // builder consumed
}
```

The existing v1 plugin path (m_gamePlugin, m_pluginRegistry) remains unchanged. When m_builder is null (v1 path), the new code is skipped entirely.

**Important:** Do NOT remove or modify the existing v1 constructor or v1 plugin flow. Both paths coexist. Phase 6 replaces the v1 path; Phase 7 removes it.

```cpp
TEST_CASE("AppBuilder integration - AddPlugin, Finalise, AppDescriptor round-trip")
{
    AppBuilder builder;

    // Plugin that registers a lifecycle hook
    struct HookPlugin : public IPlugin
    {
        static inline bool HookFired = false;
        void Build(AppBuilder& b) override
        {
            b.OnAppReady([](EngineContext&) { HookFired = true; });
        }
    };

    builder.AddPlugin<HookPlugin>();
    auto result = builder.Finalise();
    REQUIRE(result.has_value());

    auto& descriptor = *result;
    CHECK(descriptor.Has<LifecycleHookManifest>());

    auto& hooks = descriptor.Get<LifecycleHookManifest>();
    CHECK(hooks.OnAppReady.size() == 1);

    // Fire the hooks to verify they work
    EngineContext ctx;
    for (auto& hook : hooks.OnAppReady)
        hook(ctx);
    CHECK(HookPlugin::HookFired);
}
```
